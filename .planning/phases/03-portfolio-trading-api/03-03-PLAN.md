```python
"""FastAPI lifespan: DB + PriceCache + market data source startup/shutdown."""

from __future__ import annotations

import logging
import os
from contextlib import asynccontextmanager

from fastapi import FastAPI

from .db import get_watchlist_tickers, init_database, open_database, seed_defaults
from .market import PriceCache, create_market_data_source, create_stream_router

logger = logging.getLogger(__name__)


@asynccontextmanager
async def lifespan(app: FastAPI):
    """Open DB -> init -> seed -> build PriceCache -> start market source -> mount SSE."""
    if not os.environ.get("OPENROUTER_API_KEY"):
        logger.warning(
            "OPENROUTER_API_KEY not set; chat endpoint will fail in Phase 5"
        )

    db_path = os.environ.get("DB_PATH", "db/finally.db")
    conn = open_database(db_path)
    init_database(conn)
    seed_defaults(conn)

    cache = PriceCache()
    source = create_market_data_source(cache)

    tickers = get_watchlist_tickers(conn)
    await source.start(tickers)

    app.state.db = conn
    app.state.price_cache = cache
    app.state.market_source = source
    app.include_router(create_stream_router(cache))
    ...
    try:
        yield
    finally:
        await source.stop()
        conn.close()
```

    ```python
    """HTTP edge for the portfolio subsystem: GET /, POST /trade, GET /history."""

    from __future__ import annotations

    import sqlite3
    import time

    from fastapi import APIRouter, HTTPException, Query, Request

    from app.market import PriceCache

    from . import service
    from .models import (
        HistoryResponse,
        PortfolioResponse,
        TradeRequest,
        TradeResponse,
    )


    def create_portfolio_router(
        db: sqlite3.Connection,
        cache: PriceCache,
    ) -> APIRouter:
        """Build an APIRouter closed over db + cache.

        A fresh router per call mirrors `create_stream_router` (app/market/stream.py)
        and avoids duplicate routes across test-spawned apps (01-CONTEXT.md D-04).
        """
        router = APIRouter(prefix="/api/portfolio", tags=["portfolio"])

        @router.get("", response_model=PortfolioResponse)
        async def get_portfolio() -> PortfolioResponse:
            return service.get_portfolio(db, cache)

        @router.post("/trade", response_model=TradeResponse)
        async def post_trade(request: Request, req: TradeRequest) -> TradeResponse:
            try:
                response = service.execute_trade(
                    db, cache, req.ticker, req.side, req.quantity,
                )
            except service.TradeValidationError as exc:
                raise HTTPException(
                    status_code=400,
                    detail={"error": exc.code, "message": str(exc)},
                )
            # D-07: reset the 60s snapshot clock so the tick observer does not
            # double-snapshot within the next minute. Route-level (not service-
            # level) keeps execute_trade FastAPI-agnostic per 03-CONTEXT.md D-02.
            request.app.state.last_snapshot_at = time.monotonic()
            return response

        @router.get(
            "/history",
            response_model=HistoryResponse,
        )
        async def get_history(
            limit: int = Query(default=1000, ge=1, le=1000),
        ) -> HistoryResponse:
            return service.get_history(db, limit=limit)

        return router
    ```

    Notes:
    - `prefix="/api/portfolio"` so the bare `GET /api/portfolio` is `router.get("")`.
    - `tags=["portfolio"]` helps OpenAPI — not mandatory but matches FastAPI conventions.
    - `TradeValidationError` subclasses carry a `code: str` attribute. We translate to HTTP 400
      with `detail={"error": code, "message": human}` (D-03 response contract).
    - `limit` is declared via `Query(default=1000, ge=1, le=1000)` so out-of-range values produce
      422 through FastAPI's own Query validation — **do NOT** raise `HTTPException(status_code=400)`
      in the handler for param validation. The 400 path in routes.py is reserved for business-rule
      violations on POST /trade only (the single `HTTPException(status_code=400, ...)` call site).
    - The service raises; do NOT wrap in a broader `except Exception`. Project convention says
      "exception managers only when needed" — the narrow `TradeValidationError` is the only
      catch, and it maps 1:1 to a 400.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser/frontend → FastAPI `/api/portfolio/*` | Untrusted JSON body crosses the Pydantic validation wall; anything non-conforming is rejected before `execute_trade` runs |
| FastAPI route handler → service layer | Route handlers pass already-validated primitives; service treats them as trusted for shape but enforces business rules (cash, shares, ticker membership, price availability) |
| Tick loop → snapshot observer | Observer runs on the event loop; a raising observer is isolated by per-callback `try/except` (Phase 3 D-08) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-03-01 | Tampering | `routes.py::post_trade` body | mitigate | Pydantic v2 `TradeRequest` with `ConfigDict(extra="forbid")`, `Literal["buy","sell"]`, `Field(gt=0, min_length=1, max_length=10)` blocks oversized/unknown fields and invalid enums at the HTTP boundary; 422 on violation |
| T-03-03-02 | Tampering | `service.execute_trade` SQL writes | mitigate | All SQL in Plan 02 service uses `?` parameter placeholders (verified against existing `app/db/seed.py` convention); no string concatenation or f-strings in SQL; covered by `test_routes_trade.py::TestErrors` (unknown_ticker, price_unavailable) and service-level unit tests in Plan 02 |
| T-03-03-03 | Information Disclosure | 400 error `detail.message` | accept | Phase 3 is single-user localhost with no authentication surface; error messages may include the violating quantity / ticker. No PII, no secrets. Messages are human-readable for the frontend; risk is negligible |
| T-03-03-04 | Denial of Service | `/api/portfolio/history` unbounded response | mitigate | `limit: int = Query(default=1000, ge=1, le=1000)` caps row count at 1000 per request via FastAPI Query validation (422 on out-of-range, not HTTPException); FastAPI/Starlette default body size limits apply to the (empty) GET request |
