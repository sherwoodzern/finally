    ```python
    """HTTP edge for the watchlist subsystem: GET /, POST /, DELETE /{ticker}."""

    from __future__ import annotations

    import logging
    import sqlite3

    from fastapi import APIRouter, HTTPException

    from app.market import MarketDataSource, PriceCache

    from . import service
    from .models import (
        WatchlistAddRequest,
        WatchlistMutationResponse,
        WatchlistResponse,
        normalize_ticker,
    )

    logger = logging.getLogger(__name__)


    def create_watchlist_router(
        db: sqlite3.Connection,
        cache: PriceCache,
        source: MarketDataSource,
    ) -> APIRouter:
        """Build an APIRouter closed over db + cache + source (D-13).

        A fresh router per call mirrors create_stream_router and create_portfolio_router
        and avoids duplicate routes across test-spawned apps. The mutation handlers
        follow a DB-first-then-source choreography (D-11): the SQLite write is the
        source of truth, and a post-commit source-side failure is logged with
        exc_info=True but returns 200 OK — the next app restart reconciles because
        the price-update loop reads the ticker union from `watchlist` on startup.

        Idempotent no-ops return 200 with status='exists'/'not_present' (D-06, SC#4),
        NOT 409/404.
        """

        router = APIRouter(prefix="/api/watchlist", tags=["watchlist"])

        @router.get("", response_model=WatchlistResponse)
        async def get_watchlist_route() -> WatchlistResponse:
            """Return the current watchlist with live prices from the cache (WATCH-01)."""
            return service.get_watchlist(db, cache)

        @router.post("", response_model=WatchlistMutationResponse)
        async def post_watchlist_route(
            req: WatchlistAddRequest,
        ) -> WatchlistMutationResponse:
            """Add a ticker; idempotent no-op on duplicate (WATCH-02, D-06, D-09, D-11)."""
            result = service.add_ticker(db, req.ticker)

            if result.status == "added":
                try:
                    await source.add_ticker(req.ticker)
                except Exception:
                    # D-11: DB is the reconciliation anchor; next restart heals.
                    logger.warning(
                        "Watchlist: source.add_ticker(%s) raised after DB commit",
                        req.ticker,
                        exc_info=True,
                    )

            return WatchlistMutationResponse(
                ticker=result.ticker, status=result.status
            )

        @router.delete("/{ticker}", response_model=WatchlistMutationResponse)
        async def delete_watchlist_route(ticker: str) -> WatchlistMutationResponse:
            """Remove a ticker; idempotent no-op when absent (WATCH-03, D-06, D-10, D-11)."""
            try:
                normalized = normalize_ticker(ticker)
            except ValueError as exc:
                raise HTTPException(status_code=422, detail=str(exc)) from exc

            result = service.remove_ticker(db, normalized)

            if result.status == "removed":
                try:
                    await source.remove_ticker(normalized)
                except Exception:
                    logger.warning(
                        "Watchlist: source.remove_ticker(%s) raised after DB commit",
                        normalized,
                        exc_info=True,
                    )

            return WatchlistMutationResponse(
                ticker=result.ticker, status=result.status
            )

        return router
    ```

    Critical conformance:
    - `prefix="/api/watchlist"` + `tags=["watchlist"]` on `APIRouter`, not per-handler (Phase 3 parity).
    - `response_model=` on every decorator (Phase 3 parity).
    - NO `request: Request` injection (watchlist doesn't mutate `app.state.last_snapshot_at`).
    - GET has no query params, no cache-cold HTTPException (Plan 04-01's service returns None for missing prices — the response model accepts it).
    - POST handler calls `service.add_ticker(db, req.ticker)` — `req.ticker` is already normalized by the Pydantic before-validator.
    - DELETE handler calls `normalize_ticker(ticker)` FIRST on the path param, mapping `ValueError → HTTPException(422)`. Do NOT let a malformed path param reach `service.remove_ticker`.
    - `try/except Exception` around the `await source.*_ticker` call ONLY — narrow scope. Post-commit failures log WARNING with `exc_info=True` and swallow (D-11). The handler still returns 200.
    - Module-level `logger = logging.getLogger(__name__)` — needed for the two WARNING paths (the portfolio/routes.py analog lacks this because that module has no logging; we add it here).
    - `%`-style log args; NO f-strings (Pitfall 4 / CONVENTIONS).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client → FastAPI route | Untrusted ticker reaches POST body / DELETE path param. Must be normalized + regex-validated before touching SQL or the market source. |
| Route handler → MarketDataSource | Trusted. Source is constructed inside lifespan; failures are logged but do not leak. |
| Route handler → sqlite3.Connection | Trusted. All SQL parameterized in service.py (Plan 04-01). Routes do not write SQL. |
| Process startup → FastAPI app | Router registration happens in `lifespan` before `yield`. No external input crosses this boundary. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-06 | Tampering / Injection | POST `/api/watchlist` body.ticker | mitigate | Pydantic `WatchlistAddRequest` with `field_validator(mode="before")` calls `normalize_ticker`, which regex-validates `^[A-Z][A-Z0-9.]{0,9}$`. Extra keys rejected via `ConfigDict(extra="forbid")`. Malformed input → 422 before handler runs. |
| T-04-07 | Tampering / Injection | DELETE `/api/watchlist/{ticker}` path param | mitigate | Handler calls `normalize_ticker(ticker)` FIRST; `ValueError` → `HTTPException(422)`. Service never sees unnormalized input. |
| T-04-08 | Denial of Service | Unbounded watchlist growth via POST spam | accept | Single-user, localhost bind (Phase 8). Regex caps each ticker at 10 chars; table size is O(#unique valid tickers) which is bounded in practice. |
| T-04-09 | Information Disclosure | Error responses could leak internals | mitigate | Pydantic 422 default body is structured but schema-shaped; no stack traces. `HTTPException(status_code=422, detail=str(exc))` for path-param exposes only "invalid ticker: ..." — safe to leak (the ticker was user-supplied). |
| T-04-10 | Repudiation | Watchlist mutations could be unrecoverable | accept | DB is source of truth (D-11); post-commit source failure is logged with WARNING + `exc_info=True` so operators can reconcile on next restart (the price loop reads the union of tickers from `watchlist`). |
| T-04-11 | Spoofing / AuthZ | Multi-user assumptions | accept | Single-user demo; `user_id = "default"` hardcoded. Phase 2 lock D-04 and Phase 5 will carry the same constraint. |
| T-04-12 | Elevation of Privilege | Client could attempt SQL via ticker field | mitigate | Two-layer defense: (a) regex rejects `;`, `'`, `"`, `--`, any SQL metacharacter at the Pydantic boundary; (b) service.py uses only `?` parameterized queries (Plan 04-01, acceptance-criteria-verified). No string interpolation anywhere. |
| T-04-13 | Denial of Service | Lifespan crash on startup | accept | The two new lines are an import + a router include — both fail fast at import time if misconfigured, and the new `test_includes_watchlist_router_during_startup` test catches regression. |
| T-04-14 | Tampering | Developer accidentally mounts the wrong router | mitigate | `test_includes_watchlist_router_during_startup` pins the exact path strings `/api/watchlist` and `/api/watchlist/{ticker}`. CI (full pytest run) catches regression. |
| T-04-15 | Information Disclosure | Route registration noise in logs | accept | FastAPI logs route registration at DEBUG. Default INFO level does not emit. |
